110def unnormalize(img_tensor):
111    """
112    Reverses normalization on a tensor image using ImageNet stats.
113
114    Parameters
115    ----------
116    img_tensor : torch.Tensor
117        Image tensor normalized with ImageNet statistics, of shape (C, H, W).
118
119    Returns
120    -------
121    torch.Tensor
122        Unnormalized image tensor with same shape.
123    """
124    mean = torch.tensor([0.485, 0.456, 0.406]).view(3,1,1)
125    std = torch.tensor([0.229, 0.224, 0.225]).view(3,1,1)
126    return img_tensor * std + mean
